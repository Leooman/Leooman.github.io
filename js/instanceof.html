V instanceof target #

Determine whether the value of target's "prototype" property is present in V's prototype chain.

If Type(target) is not Object, throw a TypeError exception.

js
//所有javascript对象或者函数instanceof Object为true
//所有函数instanceof Function为true
function _instanceof(L,R){
    let O = R.prototype //获取R的显示原型
    L = L.__proto__ //获取L的隐式原型
    while(true){
        if(L === null) return false
        if(O === L) return true
        L = L.__proto__
    }
}
