全局环境 #

  • 浏览器环境：无论是否在严格模式下，在全局执行环境中（在任何函数体外部）this 都指向全局对象 window

  • node 环境：无论是否在严格模式下，在全局执行环境中（在任何函数体外部）this 都是空对象 {}

构造器调用 #

  • 构造函数中没有返回 function 或 object，this 指向这个新对象
js
// 无return
function Super(age) {
    this.age = age;
}
let instance = new Super('26');
console.log(instance.age); //26
// return 非object类型
var MyClass = function(){
	this.name = 'sven'
	return 'anne'; // 返回string类型
};
var obj = new MyClass();
alert ( obj.name ); // sven
  • 构造函数返回值是 function 或 object，this指向构造函数返回的对象
js
function Super(age) {
    this.age = age;
    let obj = {a: '2'};
    return obj;
}
 
let instance = new Super('hello'); 
console.log(instance);//{ a: '2' }
console.log(instance.age); //undefined

显式绑定 #

函数通过 call/apply 调用，或者使用bind绑定

js
var person = {
    age: 20,
    info:function(){
        console.log(this.age)
    }
}
var age = 28;
var info = person.info;
info.call(person);   //20
info.apply(person);  //20
info.bind(person)(); //20
  • 如果call/apply或者 bind 传入的第一个参数值是 undefined 或者 null，this 指向全局对象 (node 环境为 global，浏览器环境为 window)
js
var person = {
    age: 20,
    info:function(){
        console.log(this);
        console.log(this.age);
    }
}
var age = 28;
var info = person.info;
// 严格模式抛出错误；
// 非严格模式，node 下输出 undefined（因为全局的 age 不会挂在 global 上）
// 非严格模式。浏览器环境下输出 28（因为全局的 age 会挂在 window 上）
info.call(null);

隐式绑定 #

作为对象的方法调用，存在上下文对象

js
var person = {
    age: 20,
    info:function(){
        console.log(this.age);
    }
}
var age = 28;
person.info(); //20; 执行的是隐式绑定

默认绑定 #

作为普通函数调用

  • 非严格模式： node 环境，指向全局对象 global，浏览器环境，指向全局对象 window

  • 严格模式： undefined

js
function info(){
    console.log(this.age);
}
var age = 28;
// 严格模式；抛错
// 非严格模式，node 下输出 undefined（因为全局的 age 不会挂在 global 上）
// 非严格模式。浏览器环境下输出 28（因为全局的 age 会挂在 window 上）
// 严格模式抛出，因为 this 此时是 undefined
info();

箭头函数 #

箭头函数没有自己的 this，继承外层上下文绑定的 this

js
let obj = {
    age: 20,
    info: function() {
        return () => {
            console.log(this.age); //this 继承的是外层上下文绑定的 this
        }
    }
}
 
let person = {age: 28};
let info = obj.info();
info(); //20
 
let info2 = obj.info.call(person);
info2(); //28

IIFE #

js
//立即执行函数中的this指向window，因为立即执行函数是window调用的 
var name = 'aaa';
function fn(){
    var name = 'bbb';
    (function(){
        console.log(this) // window
        console.log(this.name) // aaa
    })()
}
fn()
js
var name = 'window'

var person1 = {
  name: 'person1',
  show1: function () {
    console.log(this.name)
  },
  show2: () => console.log(this.name),
  show3: function () {
    return function () {
      console.log(this.name)
    }
  },
  show4: function () {
    return () => console.log(this.name)
  }
}
var person2 = { name: 'person2' }

person1.show1()//person1
person1.show1.call(person2)//person2

person1.show2()//window
person1.show2.call(person2)//window

person1.show3()()//window
person1.show3().call(person2)//person2
person1.show3.call(person2)()//window

person1.show4()()//person1
person1.show4().call(person2)//person1
person1.show4.call(person2)()//person2
