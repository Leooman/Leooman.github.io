指令

description

指令可以在DOM元素上附加行为，一个指令必须属于NgModule模块，在NgModule的declarations字段中声明，以便其他的指令、组件或应用使用。

how to use

import { Directive } from '@angular/core';
@Directive({
    selector?: string,                      //=> 定义组件在HTML代码中匹配的标签
    inputs  ?: string[],                    //=> 指令的输入属性集合
    outputs ?: string[],                    //=> 指令的输出属性集合
    host    ?: {                            //=> 绑定与宿主元素相关的事件、属性
        [key: string]: string
    },
    providers?: Provider[],                 //=> 注入该指令/组件或子组件可用的服务集合
    exportAs ?: string,                     //=> 定义可在模板中使用的名称，以将此指令分配给变量
    queries  ?: {                           //=> 配置指令的查询条件
        [key: string]: any
    }
})
export class DirectiveName{
}

selector

  • element-name: 元素名称
  • .class: 元素类名
  • [attribute]: 元素属性名
  • [attribute=value]: 元素属性名/值
  • :not(sub_selector): 非匹配元素
  • selector1, selector2: 匹配 selector1 或 selector2

inputs

inputs 属性定义了[ directiveProperty : bindingProperty] 配置数组:

  • directiveProperty 指定渲染指令/组件时的属性名称，在当前指令/组件中使用
  • bindingProperty 指定调用指令/组件时传入的属性名称，在其他指令/组件调用时使用

如果bindingProperty 没有定义，和directiveProperty定义的名称相同. 

//=> bankAccount.component.ts
import { Component,Input } from "@angular/core"
@Component({
    selector: 'bank-account',
    //inputs: ['bankName', 'id: account-id'],
    template: `
      Bank Name: {{bankName}}
      Account Id: {{id}}
    `
})
export class BankAccountComponent {
    @Input() bankName: string;
    @Input('account-id') id: string;
    // this property is not bound, and won't be automatically updated by Angular
    normalizedBankName: string;
}

//=> app.component.ts
import { Component } from "@angular/core"
@Component({
    selector: 'app',
    template: `
      <bank-account [bankName]="bank" account-id="4747"></bank-account>
    `
})
export class AppComponent {
    bank : string = "RBC"
}

outputs

outputs 属性定义了[ directiveProperty : bindingProperty] 配置数组:

  • directiveProperty 指定指令/组件内部触发事件的名称
  • bindingProperty 指定调用指令/组件时附加到DOM元素上的事件名称
//=> interval.directive.ts
import { Directive,Output,EventEmitter } from "@angular/core"
@Directive({
    selector: 'interval-dir',
    //outputs: ['everySecond', 'five5Secs: everyFiveSeconds']
})
export class IntervalDirDirective {
    @Output() everySecond = new EventEmitter();
    @Output('everyFiveSeconds') five5Secs = new EventEmitter();
    constructor() {
      setInterval(() => this.everySecond.emit("event"), 1000);
      setInterval(() => this.five5Secs.emit("event"), 5000);
    }
}

//=> app.component.ts
@Component({
    selector: 'app',
    template: `
      <interval-dir (everySecond)="everySecond()" (everyFiveSeconds)="everyFiveSeconds()">
      </interval-dir>
    `
})
export class AppComponent {
    everySecond() { console.log('second'); }
    everyFiveSeconds() { console.log('five seconds'); }
}

host

  • HostListener宿主元素绑定事件 
//=> countClicks.directive.ts
import {Directive,HostListener} from "@angular/core"
@Directive({
    selector: 'button[counting]',
    //host: {
    //    '(click)': 'onClick($event.target)'
        //}
})
export class CountClicks {
    numberOfClicks = 0;

    @HostListener('click', ['$event.target'])
    onClick(btn) {
      console.log('button', btn, 'number of clicks:', this.numberOfClicks++);
    }
}

//=> app.component.ts
@Component({
    selector: 'app',
    template: '<button counting>Increment</button>',
})
export class AppComponent {}
  • HostBinding宿主元素绑定属性
//=> typeStatus.directive.ts
import { Directive,HostBinding } from "@angular/core"
@Directive({
    selector: '[ngModel]',
    //host: {
    //  'type': 'password'
    //}
})
export class TypeStatus {
constructor() {}
    @HostBinding('attr.type') type = "password"
}

//=> app.component.ts
@Component({
    selector: 'app',
    template: `<input [ngModel]>`,
})
export class AppComponent {}
//=> <input class="ng-untouched ng-pristine ng-valid" type="password">

providers

//=> logger.service.ts
import { Injectable } from "@angular/core"
@Injectable()
export class LoggerService {
    constructor(){}
    time(){
        return new Date()
    }
}

//=> time.directive.ts
@Directive({
  selector: 'time',
  providers: [
    LoggerService
  ]
})
export class TimeDirective {
    tiem:any;

    constructor(log:LoggerService){
        this.time = log.time()
        console.log(this.time)
    }
}

exportAs

//=> time.directive.ts
@Directive({
  selector: 'time',
  providers: [
    LoggerService
  ],
  exportAs:"T"
})
export class TimeDirective{
    time:any;

    constructor(log:LoggerService){
        this.time = log.time()
    }
}

//=> app.component.ts
@Component({
  selector: 'app',
  template: `<time #c="T">{{c.time}}</time>`
})
class AppComponent {
}

内容查询在ngAfterContentInit的回调函数之前执行

视图查询在ngAfterViewInit的回调函数之前执行

@Component({
  selector: 'someDir',
  queries: {
    contentChildren: new ContentChildren(ChildDirective),
    viewChildren: new ViewChildren(ChildDirective)
  },
  template: '<child-directive></child-directive>'
})
export class SomeDir {
  contentChildren: QueryList<ChildDirective>,
  viewChildren: QueryList<ChildDirective>
  ngAfterContentInit() {
    // contentChildren is set
  }
  ngAfterViewInit() {
    // viewChildren is set
  }
}
