```js
// 1 2 5组合100，有多少种方法
let count = 0
//穷举法
for(let i = 0; i <= 100; i += 5){
     for(let j = 0; j <= 100 - i; j += 2){
         for(let k = 0; k <= 100 - i - j; k++)
             if(i + j + k == 100)
                count++;
     }
}
//优化1，对1没有必要循环，只要i+j <= 100肯定存在一种解法，也就是说1肯定能补全差值
for(let i = 0; i <= 100; i += 5){
     for(let j = 0; j <= 100 - i; j += 2){
         count++;
     }
}
//最优
for(let i = 0;i <= 100;i+=5){
    count+=(100-i+2) >> 1 //只要（100-i）是2的倍数就可以，考虑到j可以为0，所以每次循环多加一次
}
```
