//给定一个随机数组，返回递增的最大长度，如[1,3,5,4,7]输出3
function fn(arr){
    let max,tmp = 0
    arr.sort((a,b) => a - b).reduce((total,current) => {
        current === total + 1 ? tmp++ : tmp=1
        if(!max || tmp > max) max = tmp
        return current
    },arr[0] - 1)
    return max
}

//二进制中1的个数
// n & (n-1)能将数字 n 的二进制表示中，最右边的 1 变成 0
function fn(n) {
    let count = 0
    while (n) {
        n = n & (n - 1)
        ++count
    }
    return count
}

//获取数组最小k个元素
//[2, 8, 1, 1, 0, 11, -1, 0]
function partiton(arr, start, end) {
    const k = arr[start];
    let left = start + 1,
        right = end;
    while (1) {
        while (left <= end && arr[left] <= k) ++left;
        while (right >= start + 1 && arr[right] >= k) --right;

        if (left >= right) {
            break;
        }

        [arr[left], arr[right]] = [arr[right], arr[left]];
        ++left;
        --right;
    }
    [arr[right], arr[start]] = [arr[start], arr[right]];
    return right;
}
var getLeastNumbers = function(arr, k) {
    const length = arr.length;
    if (k >= length) return arr;
    let left = 0,
        right = length - 1;
    let index = partiton(arr, left, right);
    while (index !== k) {
        if (index < k) {
            left = index + 1;
            index = partiton(arr, left, right);
        } else if (index > k) {
            right = index - 1;
            index = partiton(arr, left, right);
        }
    }

    return arr.slice(0, k);
};
